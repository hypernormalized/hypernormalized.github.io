Inspired by Hiroshi Ohchi's 1954 poster *Growing Forms through Function*. Ohchi, the founding art director of *IDEA* magazine, possessed a unique ability to bridge the gap between rigorous modernist geometry and a playful, almost biological sense of illustration.

In this study, I explore the idea of "functional growth." The forms are generated by a system that adheres to a strict underlying grid, yet the strokes themselves branch and taper with organic unpredictability.

let branches = [];
function setup() {
  let canvas = createCanvas(600, 600);
  canvas.parent('p5-container');
  background(245, 245, 240);
  stroke(30);
  noFill();
  let startX = width / 2;
  let startY = height / 2;
  for(let i=0; i<6; i++) {
    branches.push(new Branch(startX, startY, random(TWO_PI), 50));
  }
}

function draw() {
  for(let b of branches) {
    b.update();
    b.show();
  }
}

class Branch {
  constructor(x, y, angle, len) {
    this.x = x;
    this.y = y;
    this.angle = angle;
    this.len = len;
    this.growing = true;
    this.points = [{x: x, y: y}];
  }
  update() {
    if(!this.growing) return;
    let nextX = this.x + cos(this.angle) * 5;
    let nextY = this.y + sin(this.angle) * 5;
    this.points.push({x: nextX, y: nextY});
    this.x = nextX;
    this.y = nextY;
    this.angle += random(-0.2, 0.2);
    if(this.points.length > 30 || this.x < 0 || this.x > width || this.y < 0 || this.y > height) {
      this.growing = false;
    }
  }
  show() {
    beginShape();
    for(let p of this.points) {
      vertex(p.x, p.y);
    }
    endShape();
  }
}
</script>
